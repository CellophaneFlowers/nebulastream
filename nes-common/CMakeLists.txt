# Set Sources and Headers
add_subdirectory(src)

get_source(nes-common NES_COMMON_SOURCE_FILES)

message(STATUS "Common Source Files: ${NES_COMMON_SOURCE_FILES}")

# Add Library
add_library(nes-common SHARED ${NES_COMMON_SOURCE_FILES})
target_link_libraries(nes-common PRIVATE ${LIBRARIES})
target_include_directories(nes-common PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include/nebulastream/>)

#TODO select the headers to copy more fine-grained
register_public_header(nes-common ${CMAKE_CURRENT_SOURCE_DIR}/include/)

# Add headers from dependencies
#TODO select dependency headers more fine-grained
register_public_header(nes-common ${PATH_TO_DEB_SOURCE_CODE}/include/)

if (NES_ENABLES_TESTS)
    # Add tests with command
    add_subdirectory(tests)
    message(STATUS "Tests are enabled")
else ()
    message(STATUS "Tests are disabled")
endif ()

#Add the nes common and headers to cpack
install(TARGETS nes-common
        EXPORT NebulaStreamTargets
        COMPONENT nesLibrary
        DESTINATION lib)
